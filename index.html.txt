  <script>
    function showPage(pageId) {
      document.getElementById('dashboard').classList.add('hidden');
      document.getElementById('zimmedaran').classList.add('hidden');
      document.getElementById(pageId).classList.remove('hidden');

      if(pageId === 'zimmedaran'){
        loadZimmedaran();
      }
    }

    function loadZimmedaran() {
      fetch("https://script.google.com/macros/s/AKfycbxB8OLQc1N4jSk1VxWjQmqzXPLkAJg0uK7Uo0gflqF8pUWXOVvbt2_q1STZR-OuxGFt/exec")
      .then(res => res.json())
      .then(data => {
        const tbody = document.querySelector("#zimmedarTable tbody");
        tbody.innerHTML = "";
        data.forEach(row => {
          let tr = document.createElement("tr");
          tr.innerHTML = `
            <td>${row.Naam || ''}</td>
            <td>${row.Region || ''}</td>
            <td>${row.Contact || ''}</td>
            <td>${row["Email Id"] || ''}</td>
            <td>${row.Jamiaat || ''}</td>
            <td>${row["Madani Asatiza"] || ''}</td>
            <td>${row["Qari Sahiban"] || ''}</td>
            <td>${row["Asri Teachers"] || ''}</td>
            <td>${row["Kul Talba"] || ''}</td>
          `;
          tbody.appendChild(tr);
        });
      })
      .catch(err => console.error(err));
    }
  </script>
